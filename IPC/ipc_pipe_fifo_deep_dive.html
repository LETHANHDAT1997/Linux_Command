Pipe — ống dẫn dữ liệu một chiều
Loại IPC đơn giản nhất. Hiểu Pipe là nền tảng để hiểu hết các IPC còn lại.
Ý tưởng cốt lõi
Hãy hình dung một cái ống nước thực sự: nước chỉ chảy một chiều, bạn đổ vào một đầu và lấy ra ở đầu kia theo đúng thứ tự (FIFO). Pipe trong Linux hoạt động y hệt như vậy với dữ liệu bytes.

Kernel tạo ra một vùng đệm (buffer) trong RAM — thường 64 KB trên Linux hiện đại. Một process ghi vào đầu này, process kia đọc từ đầu kia. Kernel hoàn toàn kiểm soát việc đồng bộ.
Pipe trông như thế nào trong bộ nhớ
Code đầy đủ, chạy được — có giải thích từng dòng
pipe_basic.c — compile: gcc pipe_basic.c -o pipe_basic
#include <unistd.h>    /* pipe(), read(), write(), fork(), close() */
#include <stdio.h>     /* printf(), perror() */
#include <stdlib.h>    /* exit() */
#include <string.h>    /* strlen() */
#include <sys/wait.h>  /* wait() */

int main(void) {
    int fds[2];   /* fds[0] = đầu đọc, fds[1] = đầu ghi */

    /* ── Bước 1: Tạo pipe ──────────────────────────────── */
    if (pipe(fds) == -1) {
        perror("pipe");   /* in lỗi dạng: "pipe: Too many open files" */
        exit(1);
    }
    /* Lúc này kernel đã tạo buffer 64KB trong RAM.
       fds[0] và fds[1] là 2 file descriptor trỏ vào buffer đó. */

    /* ── Bước 2: Fork để tạo child process ─────────────── */
    pid_t pid = fork();
    if (pid == -1) { perror("fork"); exit(1); }

    if (pid == 0) {
        /* ═══ Đây là CHILD PROCESS (writer) ═══ */

        close(fds[0]);   /* Đóng đầu đọc — child không cần đọc.
                            Quan trọng: nếu không đóng, reader sẽ
                            không bao giờ thấy EOF! */

        const char *msg = "Xin chao tu child process!";
        ssize_t n = write(fds[1], msg, strlen(msg));
        /* write() copy dữ liệu từ user space vào kernel buffer.
           Nếu buffer đầy → write() sẽ BLOCK cho đến khi có chỗ. */

        if (n == -1) { perror("write"); exit(1); }
        printf("[Child]  Da ghi %zd bytes\n", n);

        close(fds[1]);   /* Đóng write-end → kernel biết không còn
                            writer nào nữa → reader sẽ thấy EOF */
        exit(0);

    } else {
        /* ═══ Đây là PARENT PROCESS (reader) ═══ */

        close(fds[1]);   /* Đóng đầu ghi — parent không cần ghi.
                            Nếu không đóng: read() sẽ block mãi mãi
                            vì vẫn còn 1 fd ghi đang mở! */

        char buf[128] = {0};
        ssize_t n = read(fds[0], buf, sizeof(buf) - 1);
        /* read() copy từ kernel buffer vào buf.
           - Nếu buffer rỗng và writer còn sống → BLOCK (chờ dữ liệu)
           - Nếu tất cả write-end đã đóng     → trả về 0 (EOF)
           - Nếu có lỗi                        → trả về -1 */

        if (n == -1) { perror("read"); exit(1); }
        if (n == 0)  { printf("[Parent] EOF — child da dong pipe\n"); }
        else         { printf("[Parent] Nhan duoc: \"%s\"\n", buf); }

        close(fds[0]);

        int status;
        wait(&status);   /* Chờ child kết thúc, tránh zombie */
        printf("[Parent] Child thoat voi code %d\n",
               WEXITSTATUS(status));
    }
    return 0;
}

[Parent] Child thoat voi code 0

Thứ tự in ra có thể đảo: child/parent chạy song song. Nhưng nội dung message luôn đúng.
Bên trong kernel: pipe thực sự hoạt động ra sao
Những hành vi blocking, EOF, PIPE_BUF — tất cả đều có lý do rõ ràng.
4 quy tắc blocking của pipe
Quy tắc 1 read() block khi buffer rỗng
Nếu không có byte nào trong buffer và vẫn còn ít nhất 1 write-end đang mở, read() ngủ cho đến khi writer ghi gì đó. Kernel không tốn CPU trong lúc chờ — scheduler đưa process vào trạng thái TASK_INTERRUPTIBLE.
Quy tắc 2 read() trả về 0 (EOF) khi tất cả write-end đã đóng
Đây là cách reader biết "không còn dữ liệu nào nữa". Lỗi cực phổ biến: quên close(fds[1]) trong parent → reader block mãi mãi vì kernel thấy vẫn còn 1 fd ghi đang mở.
Quy tắc 3 write() block khi buffer đầy
Buffer mặc định 64 KB. Nếu reader chậm không đọc kịp và writer tiếp tục ghi, đến lúc buffer đầy, write() block cho đến khi reader đọc bớt ra. Điều này tạo ra cơ chế back-pressure tự nhiên.
Quy tắc 4 write() lên pipe không có reader → SIGPIPE
Nếu tất cả read-end đã đóng mà writer vẫn ghi, kernel gửi SIGPIPE đến writer. Mặc định process bị kill. Có thể ignore signal này và kiểm tra errno == EPIPE từ write().
PIPE_BUF và tính atomic
PIPE_BUF trên Linux = 4096 bytes (4 KB). Đây là ngưỡng quan trọng:

— Nếu write(n) với n ≤ PIPE_BUF: kernel đảm bảo ghi atomic — không bị xen lẫn với write() của process khác. An toàn với nhiều writer.
— Nếu n > PIPE_BUF: kernel có thể chia nhỏ và các write() khác nhau có thể bị xen kẽ → cần lock thủ công.
Kiểm tra PIPE_BUF và kích thước buffer thực tế
#include <limits.h>   /* PIPE_BUF */
#include <fcntl.h>    /* F_GETPIPE_SZ, F_SETPIPE_SZ */

printf("PIPE_BUF = %d bytes\n", PIPE_BUF);  /* Thường là 4096 */

int fds[2];
pipe(fds);

/* Đọc kích thước buffer hiện tại của pipe */
int sz = fcntl(fds[0], F_GETPIPE_SZ);
printf("Pipe buffer size = %d bytes\n", sz);  /* Thường 65536 = 64KB */

/* Tăng kích thước buffer (cần quyền hoặc process là owner) */
fcntl(fds[1], F_SETPIPE_SZ, 1024 * 1024);  /* Tăng lên 1 MB */
/* Trên embedded: /proc/sys/fs/pipe-max-size giới hạn giá trị tối đa */
Pipe với vòng lặp đọc — cách đúng
pipe_loop.c — đọc cho đến EOF, xử lý từng chunk
/* Reader đúng cách: đọc trong vòng lặp */
char buf[4096];
ssize_t n;

while ((n = read(fds[0], buf, sizeof(buf))) > 0) {
    /* n bytes đã đọc được, xử lý [buf, buf+n) */
    fwrite(buf, 1, n, stdout);
}
/* Sau vòng lặp: n == 0 nghĩa là EOF (tất cả writer đã đóng)
                 n == -1 nghĩa là lỗi, kiểm tra errno */
if (n == -1) perror("read");


/* ─── Sai lầm thường gặp ─────────────────────────────── */
/* SAI: đọc 1 lần, giả sử lấy hết dữ liệu               */
read(fds[0], buf, sizeof(buf));   /* Có thể nhận ít hơn! */

/* ĐÚNG: luôn đọc trong vòng lặp đến khi n == 0 hoặc lỗi */
Lý do phải đọc trong vòng lặp: read() được phép trả về ít hơn số byte bạn yêu cầu ngay cả khi chưa EOF. Ví dụ: write 1000 bytes nhưng read() có thể trả về 512, rồi 488. Đây là hành vi hợp lệ hoàn toàn.
API chi tiết — từng tham số, từng flag
Giải thích đầy đủ pipe(), pipe2(), read(), write() trong ngữ cảnh pipe.
pipe() — tạo anonymous pipe
#include <unistd.h>
int pipe(int pipefd[2]);
pipefd[2]
Output: kernel điền vào mảng 2 phần tử.
• pipefd[0] = file descriptor đầu đọc (read end)
• pipefd[1] = file descriptor đầu ghi (write end)
Cả hai là fd thông thường, dùng được với read/write/close/select/poll/epoll.
Trả về: 0 khi thành công, -1 khi lỗi (errno được set: EMFILE = hết fd, ENFILE = system limit)
pipe2() — tạo pipe với flags (Linux-specific)
#include <fcntl.h> /* O_CLOEXEC, O_NONBLOCK */ #include <unistd.h> int pipe2(int pipefd[2], int flags);
pipefd[2]
Giống pipe() — kernel điền đầu đọc/ghi.
flags
Kết hợp bằng OR (|) các giá trị sau:
O_CLOEXECTự động đóng fd khi process gọi exec(). Luôn dùng flag này để tránh fd rò rỉ sang chương trình con. Tương đương gọi fcntl(fd, F_SETFD, FD_CLOEXEC) nhưng atomic.
O_NONBLOCKBiến read/write thành non-blocking. Thay vì block, hàm trả về -1 với errno=EAGAIN khi không có dữ liệu (read) hoặc buffer đầy (write). Dùng khi kết hợp với event loop (select/poll/epoll).
Trả về: 0 thành công, -1 lỗi. Lỗi thêm: EINVAL nếu flags không hợp lệ
Cách dùng pipe2() đúng chuẩn cho embedded/production
int fds[2];

/* pipe2 = pipe + O_CLOEXEC + O_NONBLOCK trong 1 syscall,
   tránh race condition giữa pipe() và fcntl() */
if (pipe2(fds, O_CLOEXEC | O_NONBLOCK) == -1) {
    perror("pipe2");
    return -1;
}

/* Đọc non-blocking */
char buf[256];
ssize_t n = read(fds[0], buf, sizeof(buf));
if (n == -1) {
    if (errno == EAGAIN || errno == EWOULDBLOCK) {
        /* Bình thường: không có dữ liệu lúc này */
    } else {
        perror("read");  /* Lỗi thực sự */
    }
}
write() và read() trong ngữ cảnh pipe
#include <unistd.h> ssize_t write(int fd, const void *buf, size_t count); ssize_t read (int fd, void *buf, size_t count);
fd
Với pipe: fds[1] cho write, fds[0] cho read. Dùng sai (ghi vào [0] hoặc đọc từ [1]) → EBADF.
buf
Con trỏ đến vùng nhớ chứa dữ liệu (write) hoặc nơi nhận dữ liệu (read). Phải valid và đủ kích thước.
count
Số bytes muốn ghi/đọc. Kết quả thực tế có thể ít hơn — luôn kiểm tra giá trị trả về.
Trả về: số bytes thực sự đọc/ghi (≥0), hoặc -1 lỗi. Riêng read: 0 = EOF
Partial write: write() có thể ghi ít hơn count bytes, đặc biệt với pipe lớn hoặc signal ngắt. Phải kiểm tra giá trị trả về và ghi lại phần còn thiếu — đây là lỗi rất phổ biến trong code production.
write_all() — helper function viết đủ bytes
/* Đảm bảo ghi đủ n bytes, xử lý partial write và EINTR */
ssize_t write_all(int fd, const void *buf, size_t n) {
    const char *p = buf;
    size_t remaining = n;
    while (remaining > 0) {
        ssize_t w = write(fd, p, remaining);
        if (w == -1) {
            if (errno == EINTR) continue;  /* Signal ngắt, thử lại */
            return -1;                     /* Lỗi thực sự */
        }
        p         += w;
        remaining -= w;
    }
    return (ssize_t)n;
}
FIFO — Named Pipe, có tên trên filesystem
Giống anonymous pipe nhưng có path, dùng được giữa các process không liên quan.
Điểm khác biệt so với anonymous pipe
Anonymous pipe
— Không có tên trên filesystem
— Chỉ dùng giữa parent và child (kế thừa fd qua fork)
— Mất khi process chết
— Tạo bằng pipe()
FIFO (Named pipe)
— Có inode trên filesystem (thường ở /tmp)
— Bất kỳ process nào biết path đều dùng được
— Tồn tại cho đến khi bị unlink()
— Tạo bằng mkfifo()
mkfifo() — tạo FIFO
#include <sys/types.h> #include <sys/stat.h> int mkfifo(const char *pathname, mode_t mode);
pathname
Đường dẫn tạo FIFO trên filesystem. Ví dụ: "/tmp/my_fifo" hoặc "/var/run/myapp/cmd". Thư mục cha phải tồn tại.
mode
Permission bits, giống chmod. Ví dụ: 0666 = đọc/ghi cho tất cả. Giá trị thực tế bị mask bởi umask của process. Trong embedded thường dùng 0600 (chỉ owner đọc/ghi) hoặc 0660 (owner + group).
Trả về: 0 thành công, -1 lỗi. Lỗi quan trọng: EEXIST = path đã tồn tại, ENOENT = thư mục cha không tồn tại
Hành vi quan trọng: open() FIFO block cho đến khi cả 2 đầu sẵn sàng
Đây là điểm khác biệt lớn nhất so với anonymous pipe. Khi mở FIFO:

— open(path, O_RDONLY): block cho đến khi có process mở O_WRONLY
— open(path, O_WRONLY): block cho đến khi có process mở O_RDONLY

Lý do: kernel cần cả 2 đầu sẵn sàng mới có ý nghĩa. Nếu muốn không block: dùng O_NONBLOCK.
fifo_server.c — bên nhận lệnh (chạy trước)
#include <sys/stat.h>  /* mkfifo() */
#include <fcntl.h>     /* open(), O_RDONLY */
#include <unistd.h>    /* read(), close(), unlink() */
#include <stdio.h>

#define FIFO_PATH "/tmp/cmd_fifo"

int main(void) {
    /* Tạo FIFO nếu chưa có */
    if (mkfifo(FIFO_PATH, 0660) == -1) {
        if (errno != EEXIST) {   /* EEXIST = đã tồn tại, OK */
            perror("mkfifo"); return 1;
        }
    }
    printf("Dang cho lenh... (FIFO: %s)\n", FIFO_PATH);

    /* open() sẽ BLOCK ở đây cho đến khi client mở để ghi */
    int fd = open(FIFO_PATH, O_RDONLY);
    if (fd == -1) { perror("open"); return 1; }

    char buf[256];
    ssize_t n;
    while ((n = read(fd, buf, sizeof(buf)-1)) > 0) {
        buf[n] = '\0';
        printf("Nhan lenh: [%s]\n", buf);
    }
    close(fd);

    /* Dọn dẹp — FIFO vẫn tồn tại trên disk nếu không unlink */
    unlink(FIFO_PATH);
    return 0;
}
fifo_client.c — bên gửi lệnh (chạy sau)
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

#define FIFO_PATH "/tmp/cmd_fifo"

int main(void) {
    /* open() block cho đến khi server mở O_RDONLY */
    int fd = open(FIFO_PATH, O_WRONLY);
    if (fd == -1) { perror("open"); return 1; }

    const char *cmd = "START_SENSOR";
    write(fd, cmd, strlen(cmd));
    printf("Da gui: %s\n", cmd);

    close(fd);  /* Đóng → server thấy EOF */
    return 0;
}

Pipe & FIFO trong Embedded Linux
Những điểm khác biệt, ưu tiên, và giới hạn khi dùng trên thiết bị nhúng.
Những ràng buộc đặc thù của embedded Linux
RAM giới hạn: Buffer mặc định 64 KB/pipe. Trên thiết bị 64 MB RAM, tạo nhiều pipe có thể chiếm đáng kể tài nguyên. Kiểm tra: cat /proc/sys/fs/pipe-max-size
Filesystem read-only: FIFO cần ghi vào filesystem. Trên hệ thống có rootfs read-only (SquashFS + overlayfs), phải tạo FIFO ở /tmp (tmpfs) hoặc /var/run (ramfs), không phải /etc hay /usr.
Ít fd hơn: ulimit -n trên embedded thường là 256–1024. Pipe tốn 2 fd. Luôn close() fd không cần thiết ngay sau fork().
Không có /proc/sys/fs/pipe-max-size trên kernel cũ: Buildroot với kernel < 3.5 có thể không có F_GETPIPE_SZ / F_SETPIPE_SZ. Kiểm tra bằng uname -r trước khi dùng.
SIGPIPE có thể kill daemon: Nếu daemon ghi vào pipe mà reader đã chết, SIGPIPE kill daemon. Luôn ignore hoặc handle SIGPIPE trong daemon code.
Pattern thực tế: pipe giữa main task và sensor task
embedded_pipe.c — pattern dùng pipe + select() để không block main loop
#include <unistd.h>
#include <fcntl.h>
#include <signal.h>
#include <sys/select.h>

/* Ignore SIGPIPE — quan trọng cho daemon */
signal(SIGPIPE, SIG_IGN);

int fds[2];
/* O_CLOEXEC: tự đóng khi exec — bắt buộc trên embedded
   để tránh rò fd khi spawn subprocess (busybox, etc.) */
pipe2(fds, O_CLOEXEC | O_NONBLOCK);

/* ── Sensor thread ghi kết quả ── */
void sensor_thread(void) {
    float temperature = read_sensor();
    /* Ghi struct nhỏ hơn PIPE_BUF → atomic, an toàn */
    write(fds[1], &temperature, sizeof(float));
}

/* ── Main loop đọc không blocking bằng select() ── */
void main_loop(void) {
    fd_set rfds;
    struct timeval tv = { .tv_sec = 0, .tv_usec = 10000 }; /* 10ms */

    FD_ZERO(&rfds);
    FD_SET(fds[0], &rfds);

    int ret = select(fds[0] + 1, &rfds, NULL, NULL, &tv);
    if (ret > 0 && FD_ISSET(fds[0], &rfds)) {
        float temp;
        if (read(fds[0], &temp, sizeof(float)) == sizeof(float)) {
            handle_temperature(temp);
        }
    }
    /* ret == 0 → timeout, không có data, tiếp tục loop bình thường */
}
Khi nào dùng Pipe vs FIFO trong embedded
Dùng Pipe khi
— Giao tiếp giữa parent và child process (sau fork)
— Giao tiếp giữa các thread trong cùng process
— Cần self-pipe trick để wake up select/epoll từ signal handler
— Không muốn để lại vết trên filesystem
Dùng FIFO khi
— Hai daemon độc lập cần giao tiếp (không có quan hệ cha-con)
— Cần giao diện đơn giản: echo "cmd" > /tmp/fifo
— Script shell cần giao tiếp với C daemon
— Debugging: có thể cat /tmp/fifo để xem data
Self-pipe trick — wake epoll từ signal handler
Pattern quan trọng trong embedded daemon: signal handler không được gọi non-async-safe functions, nhưng bạn cần main loop phản ứng với SIGTERM/SIGUSR1. Giải pháp: pipe 1 byte vào một pipe, epoll sẽ thức dậy.
self_pipe_trick.c
static int wakeup_pipe[2];  /* global */

/* Signal handler — chỉ write() 1 byte, là async-signal-safe */
void sig_handler(int sig) {
    char byte = (char)sig;
    write(wakeup_pipe[1], &byte, 1);  /* Không check return — intentional */
}

int main(void) {
    pipe2(wakeup_pipe, O_CLOEXEC | O_NONBLOCK);

    struct sigaction sa = { .sa_handler = sig_handler };
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGUSR1, &sa, NULL);

    /* Thêm wakeup_pipe[0] vào epoll */
    epoll_ctl(epfd, EPOLL_CTL_ADD, wakeup_pipe[0], &ev);

    /* Khi nhận signal → pipe có data → epoll wakeup → đọc byte → xử lý */
    /* Đây là pattern chuẩn cho event-driven daemon trên Linux */
}
