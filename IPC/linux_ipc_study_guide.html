Linux IPC — Tài liệu nghiên cứu

Inter-Process Communication: các cơ chế để các tiến trình trao đổi dữ liệu và đồng bộ hoá trên Linux kernel.

Ba nhóm IPC chính: Data Transfer (sao chép dữ liệu qua kernel), Shared Memory (chia sẻ vùng nhớ), Signaling (thông báo sự kiện)
Data Transfer
Dữ liệu được sao chép qua kernel buffer. Kernel quản lý toàn bộ việc đọc/ghi, đảm bảo đồng bộ ngầm định.

Shared Memory
Các tiến trình map cùng một vùng RAM vật lý. Không có copy overhead — bandwidth cao nhất. Cần tự đồng bộ bằng semaphore hoặc mutex.

Signaling
Thông báo sự kiện không mang dữ liệu lớn. Signal là asynchronous; semaphore/futex dùng để đồng bộ hoá truy cập tài nguyên.

Anonymous Pipe pipe(2)
Tồn tại trong RAM, không có tên trong filesystem. Chỉ dùng giữa các tiến trình có quan hệ cha-con (chia sẻ fd qua fork).
pipe(fds) — tạo 2 file descriptor
fds[0] = đầu đọc, fds[1] = đầu ghi. Kernel tạo buffer trong RAM.
fork() — nhân bản tiến trình
Child kế thừa cả hai fd. Mỗi process đóng fd không dùng.
write(fds[1]) / read(fds[0])
Kernel copy dữ liệu vào buffer. read() block cho đến khi có dữ liệu.
EOF khi writer đóng fds[1]
read() trả về 0. Reader thoát vòng lặp.
Named Pipe (FIFO) mkfifo(3)
Có inode trong filesystem, tồn tại độc lập với process. Bất kỳ process nào biết path đều có thể mở.
mkfifo("/tmp/myfifo", 0666)
Mở FIFO sẽ block cho đến khi cả hai phía (reader và writer) đều mở. Dùng O_NONBLOCK để tránh deadlock.
Code: Anonymous Pipe
pipe_example.c
#include <unistd.h>
#include <stdio.h>

int main() {
    int fds[2];
    pipe(fds);          /* tạo pipe */

    if (fork() == 0) {
        /* Child: writer */
        close(fds[0]);  /* đóng đầu đọc */
        char msg[] = "Hello from child";
        write(fds[1], msg, sizeof(msg));
        close(fds[1]);
    } else {
        /* Parent: reader */
        close(fds[1]);  /* đóng đầu ghi */
        char buf[64];
        int n = read(fds[0], buf, sizeof(buf));
        printf("Got: %.*s\n", n, buf);
        close(fds[0]);
    }
    return 0;
}

Unix Domain Socket là gì?
Socket dùng filesystem path thay vì IP:port. Giao tiếp qua kernel buffer nội bộ — không đi qua network stack. Hỗ trợ truyền file descriptor giữa các process (fd passing).
Server: socket() → bind() → listen()
Tạo socket file tại /tmp/server.sock, lắng nghe kết nối.
Client: socket() → connect()
Kết nối đến path. Kernel tạo kênh liên lạc trong RAM.
Server: accept() → trả về fd mới
accept() block cho đến khi có client. Mỗi client có fd riêng.
send() / recv() hoặc read() / write()
Bidirectional — cả hai phía đều có thể gửi và nhận.
SOCK_STREAM vs SOCK_DGRAM:
Stream = byte stream như TCP, đảm bảo thứ tự.
Dgram = message-based như UDP, giữ message boundary.
Code: Unix Domain Socket
server.c (rút gọn)
#include <sys/socket.h>
#include <sys/un.h>

#define SOCK_PATH "/tmp/demo.sock"

int main() {
    int srv = socket(AF_UNIX,
                     SOCK_STREAM, 0);
    struct sockaddr_un addr = {
        .sun_family = AF_UNIX,
        .sun_path   = SOCK_PATH
    };
    bind(srv, (struct sockaddr*)&addr,
         sizeof(addr));
    listen(srv, 5);

    int cli = accept(srv, NULL, NULL);
    char buf[256];
    int n = recv(cli, buf, sizeof(buf), 0);
    printf("Got: %.*s\n", n, buf);
    send(cli, "ack", 3, 0);

    close(cli); close(srv);
    unlink(SOCK_PATH);
}
client.c (rút gọn)
int main() {
    int fd = socket(AF_UNIX,
                    SOCK_STREAM, 0);
    struct sockaddr_un addr = {
        .sun_family = AF_UNIX,
        .sun_path   = "/tmp/demo.sock"
    };
    connect(fd, (struct sockaddr*)&addr,
            sizeof(addr));
    send(fd, "hello", 5, 0);

    char buf[16];
    recv(fd, buf, sizeof(buf), 0);
    close(fd);
}

POSIX Shared Memory shm_open
producer.c
#include <sys/mman.h>
#include <fcntl.h>

#define SHM_NAME "/myshm"
#define SIZE     4096

int main() {
    /* Tạo / mở shared memory object */
    int fd = shm_open(SHM_NAME,
                      O_CREAT|O_RDWR, 0666);
    ftruncate(fd, SIZE);

    /* Map vào address space */
    char *ptr = mmap(NULL, SIZE,
                     PROT_READ|PROT_WRITE,
                     MAP_SHARED, fd, 0);

    /* Ghi dữ liệu — process B đọc ngay */
    snprintf(ptr, SIZE, "Hello SHM!");

    /* Cleanup */
    munmap(ptr, SIZE);
    close(fd);
    /* shm_unlink(SHM_NAME); khi done */
}
Đồng bộ hoá bắt buộc

#include <semaphore.h>

/* Đặt semaphore trong SHM */
typedef struct {
    sem_t   sem;
    char    data[1024];
} SharedData;

SharedData *shm = mmap(...);
sem_init(&shm->sem, 1, 1); /* pshared=1 */

/* Producer */
sem_wait(&shm->sem);
memcpy(shm->data, buf, len);
sem_post(&shm->sem);

/* Consumer */
sem_wait(&shm->sem);
memcpy(local, shm->data, len);
sem_post(&shm->sem);

POSIX Message Queue mq_open
Hàng đợi có thứ tự ưu tiên, message-based (không phải byte stream). Kernel lưu trữ — sender gửi xong tiếp tục, receiver nhận khi rảnh.
sender.c
#include <mqueue.h>

int main() {
    mqd_t mq = mq_open("/myqueue",
        O_CREAT|O_WRONLY,
        0666, NULL);

    struct {
        int  type;
        char payload[128];
    } msg = { .type = 1 };

    snprintf(msg.payload, 128,
             "Hello MQ");

    /* priority = 5 */
    mq_send(mq, (char*)&msg,
            sizeof(msg), 5);
    mq_close(mq);
}

Code: sigaction + signalfd
sigaction — cách đúng để handle signal
#include <signal.h>
#include <stdio.h>

volatile sig_atomic_t running = 1;

void handle_term(int sig) {
    running = 0;  /* async-signal-safe */
}

int main() {
    struct sigaction sa = {
        .sa_handler = handle_term,
        .sa_flags   = SA_RESTART
    };
    sigemptyset(&sa.sa_mask);
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGINT,  &sa, NULL);

    while (running) {
        /* main loop */
        pause(); /* sleep until signal */
    }
    printf("Shutting down gracefully\n");
}
signalfd — đọc signal như file (dùng với epoll)
#include <sys/signalfd.h>

sigset_t mask;
sigemptyset(&mask);
sigaddset(&mask, SIGTERM);
sigaddset(&mask, SIGUSR1);

/* Block signals thông thường */
sigprocmask(SIG_BLOCK, &mask, NULL);

/* Tạo fd cho signals */
int sfd = signalfd(-1, &mask, SFD_CLOEXEC);

/* Đọc khi có signal */
struct signalfd_siginfo info;
read(sfd, &info, sizeof(info));
printf("Got signal %d from PID %d\n",
       info.ssi_signo, info.ssi_pid);
Lưu ý async-signal safety
Trong signal handler chỉ được gọi async-signal-safe functions. Không được: malloc, printf, mutex lock. Được: write(), _exit(), sig_atomic_t, self-pipe trick.
